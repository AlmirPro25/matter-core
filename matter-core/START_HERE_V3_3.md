### **3. Spintronics Example**
```matter
import "matter-spintronics" as spin

fn main() {
    # Create processor
    let proc = spin.SpintronicProcessor.new(1024)
    
    # Add gates
    proc.add_gate("not1", spin.SpinGateType.NOT)
    
    # Execute
    let result = proc.execute_gate("not1", [spin.SpinState.up()])
    
    print("Result: " + result.projection())
}
```
